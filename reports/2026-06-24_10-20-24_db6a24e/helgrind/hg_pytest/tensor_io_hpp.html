259    {
260      auto copy = tensor.copy();
261      if (!copy.is_contiguous())
262      {
263        // To avoid infinite loop
264        throw std::runtime_error("Tensor copy is non-contiguous/non-zero-offset (this is a bug, please report it!).");
265      }
266      write_tensor(os, copy);
267      return;
268    }
269    write_contiguous_tensor(os, tensor);
270  }

274  template <typename T>
275  Tensor<T> read_tensor(std::istream& is)
276  {
277    auto shapeSz = read_bytes<std::uint64_t>(is);
278    std::vector<size_t> shape(shapeSz);
279    std::vector<ptrdiff_t> strides(shapeSz);
280    for (auto i = 0_uz; i < shapeSz; ++i)
281    {
282      shape[i] = read_bytes<std::uint64_t>(is);
283      strides[i] = static_cast<ptrdiff_t>(read_bytes<std::uint64_t>(is));
284    }
285
286    Tensor<T> ret(shape);
287    if (ret.strides() != strides)
288    {
289      throw std::runtime_error("Incorrect strides in saved data (expected " + index_to_string(ret.strides()) + " but got " + index_to_string(strides) + ")");
290    }
291
292    auto sz = ret.size();
293    for (auto * elem = ret.data(); elem != ret.data() + sz; ++elem)
294    {
295      if constexpr (is_barcode_v<T>)
296        *elem = read_barcode<typename is_barcode<T>::scalar_type>(is);
297      else if constexpr (is_compressed_matrix_v<T>)
298        *elem = read_compressed_matrix<T>(is);
299      else
300        *elem = read_element<T>(is);
301    }
302
303    return ret;
304  }

306  // Read the current (baseFormat 1001) point cloud tensor format: distinct source
307  // coordinate buffers stored once, then per-element (source id, optional indices).
308  // Elements that reference the same source share its coordinate buffer.
309  template <typename ScalarT>
310  Tensor<PointCloud<ScalarT>> read_indexed_point_cloud_tensor(std::istream& is)
311  {
312    auto shapeSz = read_bytes<std::uint64_t>(is);
313    std::vector<size_t> shape(shapeSz);
314    std::vector<ptrdiff_t> strides(shapeSz);
315    for (auto i = 0_uz; i < shapeSz; ++i)
316    {
317      shape[i] = read_bytes<std::uint64_t>(is);
318      strides[i] = static_cast<ptrdiff_t>(read_bytes<std::uint64_t>(is));
319    }
320
321    Tensor<PointCloud<ScalarT>> ret(shape);
322    if (ret.strides() != strides)
323    {
324      throw std::runtime_error("Incorrect strides in saved data (expected " + index_to_string(ret.strides()) + " but got " + index_to_string(strides) + ")");
325    }
326
327    auto numSources = read_bytes<std::uint64_t>(is);
328    std::vector<Tensor<ScalarT>> sources;
329    sources.reserve(numSources);
330    for (auto i = 0_uz; i < numSources; ++i)
331      sources.push_back(read_element<Tensor<ScalarT>>(is));
332
333    auto sz = ret.size();
334    for (auto * elem = ret.data(); elem != ret.data() + sz; ++elem)
335    {
336      auto id = read_bytes<std::uint64_t>(is);
337      auto indexed = read_bytes<std::uint8_t>(is);
338      if (indexed)
339        *elem = PointCloud<ScalarT>(sources[id], read_element<Tensor<uint64_t>>(is));
340      else
341        *elem = PointCloud<ScalarT>(sources[id]);
342    }
343
344    return ret;
345  }
