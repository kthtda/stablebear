112  template <typename PcfT, typename UnaryF, typename MaxOp>
113  auto max_element(const Tensor<PcfT>& in, size_t dim, UnaryF&& f, MaxOp&& maxOp, Executor& exec = default_executor())
114  {
115    using OutValueT = std::decay_t<decltype(f(std::declval<PcfT>()))>;
116
117    static_assert(std::invocable<MaxOp, OutValueT, OutValueT>);
118
119    // max has no identity element, so it cannot reduce an empty range: the
120    // body below seeds from *first and reduces [first + 1, last), both UB when
121    // the slice is empty (inDimSize == 0). Reject up front with a clean,
122    // catchable exception (pybind11 maps std::invalid_argument to a Python
123    // ValueError) instead of dereferencing a past-the-end iterator -- mirrors
124    // numpy's "zero-size array to reduction operation" error. Validate dim
125    // first so shape[dim] is in range.
126    check_reduce_dim(dim, in.shape().size());
127    if (in.shape()[dim] == 0)
128    {
129      std::ostringstream oss;
130      oss << "Cannot reduce an empty dimension: dimension " << dim << " has size 0";
131      throw std::invalid_argument(oss.str());
132    }
133
134    auto fn = std::forward<UnaryF>(f);
135    auto mop = std::forward<MaxOp>(maxOp);
136    return parallel_dim_reduce<OutValueT>(in, dim, [fn, mop](auto first, auto last) {
137      auto init = fn(*first);
138      return std::transform_reduce(first + 1, last, init, mop, fn);
139    }, exec);
