354  /**
355   * Compute the broadcast-compatible output shape from two input shapes (NumPy rules).
356   *
357   * Shapes are compared right-to-left. Dimensions match if they are equal or one of them is 1.
358   * Missing leading dimensions are treated as size 1.
359   *
360   * @param a shape of the first operand
361   * @param b shape of the second operand
362   * @return the broadcast output shape
363   * @throws std::invalid_argument if the shapes are not broadcast-compatible
364   */
365  inline std::vector<size_t> broadcast_shapes(
366    const std::vector<size_t>& a,
367    const std::vector<size_t>& b)
368  {
369    size_t ndim = std::max(a.size(), b.size());
370    std::vector<size_t> result(ndim);
371
372    for (size_t i = 0; i < ndim; ++i)
373    {
374      size_t da = (i < ndim - a.size()) ? 1 : a[i - (ndim - a.size())];
375      size_t db = (i < ndim - b.size()) ? 1 : b[i - (ndim - b.size())];
376
377      if (da == db)
378        result[i] = da;
379      else if (da == 1)
380        result[i] = db;
381      else if (db == 1)
382        result[i] = da;
383      else
384        throw std::invalid_argument("Shapes are not broadcast-compatible: " +
385            shape_to_string(a) + " and " + shape_to_string(b));
386    }
387    return result;
388  }
