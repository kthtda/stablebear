517  template <typename T>
518  Tensor<T> masked_select(const Tensor<T>& src, const Tensor<bool>& mask)
519  {
520    if (mask.shape() != src.shape())
521    {
522      throw std::invalid_argument(
523        "masked_select: mask shape does not match tensor shape");
524    }
525
526    // Single pass: collect matching elements into a buffer
527    std::vector<T> buf;
528    walk(src, [&](const std::vector<size_t>& idx) {
529      if (mask(idx))
530        buf.push_back(src(idx));
531    });
532
533    if (buf.empty())
534    {
535      return Tensor<T>({0});
536    }
537
538    Tensor<T> result({buf.size()});
539    std::move(buf.begin(), buf.end(), result.data());
540
541    return result;
542  }

545  void masked_assign(Tensor<T>& dst, const Tensor<bool>& mask, const Tensor<T>& values)
546  {

772    struct ResolvedSelector
773    {
774      size_t axis;
775      std::vector<size_t> indices;  // selected positions along this axis
776    };
777
778    inline std::vector<ResolvedSelector> resolve_selectors(
779      const std::vector<std::pair<size_t, AxisSelector>>& selectors,
780      const std::vector<size_t>& shape)
781    {
782      std::vector<ResolvedSelector> result;
783      result.reserve(selectors.size());
784      for (const auto& [axis, sel] : selectors)
785      {
786        if (axis >= shape.size())
787          throw std::invalid_argument("outer indexing: axis " + std::to_string(axis) +
788            " out of range for tensor with " + std::to_string(shape.size()) + " dimensions");
789
790        ResolvedSelector rs;
791        rs.axis = axis;
792        std::visit([&](const auto& idx_tensor) {
793          using IdxT = std::decay_t<decltype(idx_tensor)>;
794          if (idx_tensor.shape().size() != 1)
795            throw std::invalid_argument("outer indexing: selector must be 1D");
796
797          if constexpr (std::is_same_v<IdxT, Tensor<bool>>)
798          {
799            if (idx_tensor.shape()[0] != shape[axis])
800              throw std::invalid_argument("outer indexing: bool mask length must match axis size");
801            rs.indices = collect_true_indices(idx_tensor);
802          }
803          else
804          {
805            const auto axis_size = static_cast<ptrdiff_t>(shape[axis]);
806            rs.indices.reserve(idx_tensor.shape()[0]);
807            for (size_t i = 0; i < idx_tensor.shape()[0]; ++i)
808            {
809              // Resolve negatives against the axis size and bounds-check so
810              // the C++ entry point is memory-safe regardless of caller.
811              auto v = static_cast<ptrdiff_t>(idx_tensor({i}));
812              if (v < 0) v += axis_size;
813              if (v < 0 || v >= axis_size)
814                throw std::out_of_range(
815                  "index " + std::to_string(static_cast<ptrdiff_t>(idx_tensor({i}))) +
816                  " is out of bounds for axis with size " + std::to_string(axis_size));
817              rs.indices.push_back(static_cast<size_t>(v));
818            }
819          }
820        }, sel);
821        result.push_back(std::move(rs));
822      }
823      return result;
824    }
825  }

931  template <typename T>
932  Tensor<T> concatenate(const std::vector<Tensor<T>>& tensors, size_t axis)
933  {
934    if (tensors.empty())
935      throw std::invalid_argument("concatenate: need at least one tensor");
936
937    auto ndim = tensors[0].shape().size();
938    if (axis >= ndim)
939      throw std::invalid_argument("concatenate: axis " + std::to_string(axis) +
940        " out of range for " + std::to_string(ndim) + "-D tensor");
941
942    // Validate shapes match on all axes except the join axis
943    for (size_t t = 1; t < tensors.size(); ++t)
944    {
945      if (tensors[t].shape().size() != ndim)
946        throw std::invalid_argument("concatenate: all tensors must have the same number of dimensions");
947      for (size_t d = 0; d < ndim; ++d)
948      {
949        if (d != axis && tensors[t].shape()[d] != tensors[0].shape()[d])
950          throw std::invalid_argument("concatenate: shape mismatch on dimension " + std::to_string(d));
951      }
952    }
953
954    // Compute output shape
955    auto out_shape = tensors[0].shape();
956    out_shape[axis] = 0;
957    for (const auto& t : tensors)
958      out_shape[axis] += t.shape()[axis];
959
960    Tensor<T> result(out_shape);
961
962    // Copy each tensor's data into the result
963    size_t offset = 0;
964    for (const auto& src : tensors)
965    {
966      auto src_axis_size = src.shape()[axis];
967      walk(src, [&](const std::vector<size_t>& src_idx) {
968        auto dst_idx = src_idx;
969        dst_idx[axis] += offset;
970        result(dst_idx) = src(src_idx);
971      });
972      offset += src_axis_size;
973    }
974
975    return result;
976  }

978  template <typename T>
979  Tensor<T> stack(const std::vector<Tensor<T>>& tensors, ptrdiff_t axis)
980  {
981    if (tensors.empty())
982      throw std::invalid_argument("stack: need at least one tensor");
983
984    auto ndim = static_cast<ptrdiff_t>(tensors[0].shape().size());
985
986    // Resolve negative axis; valid range for new axis is [-(ndim+1), ndim]
987    if (axis < 0)
988      axis += ndim + 1;
989    if (axis < 0 || axis > ndim)
990      throw std::invalid_argument("stack: axis out of range");
991
992    // Validate all shapes match
993    for (size_t t = 1; t < tensors.size(); ++t)
994    {
995      if (tensors[t].shape() != tensors[0].shape())
996        throw std::invalid_argument("stack: all tensors must have the same shape");
997    }
998
999    // expand_dims each tensor, then concatenate along the new axis
1000    std::vector<Tensor<T>> expanded;
1001    expanded.reserve(tensors.size());
1002    for (const auto& t : tensors)
1003      expanded.push_back(t.expand_dims(axis));
1004
1005    return concatenate(expanded, static_cast<size_t>(axis));
1006  }

1008  template <typename T>
1009  std::vector<Tensor<T>> split(const Tensor<T>& tensor,
1010    const std::vector<size_t>& split_points, size_t axis)
1011  {
1012    auto ndim = tensor.shape().size();
1013    if (axis >= ndim)
1014      throw std::invalid_argument("split: axis " + std::to_string(axis) +
1015        " out of range for " + std::to_string(ndim) + "-D tensor");
1016
1017    auto axis_size = tensor.shape()[axis];
1018
1019    // Build slice boundaries: [0, sp0, sp1, ..., axis_size]
1020    std::vector<size_t> boundaries;
1021    boundaries.reserve(split_points.size() + 2);
1022    boundaries.push_back(0);
1023    for (auto sp : split_points)
1024    {
1025      if (sp > axis_size)
1026        sp = axis_size;
1027      boundaries.push_back(sp);
1028    }
1029    boundaries.push_back(axis_size);
1030
1031    std::vector<Tensor<T>> result;
1032    result.reserve(boundaries.size() - 1);
1033    for (size_t i = 0; i + 1 < boundaries.size(); ++i)
1034    {
1035      std::vector<Slice> slices(ndim, all());
1036      slices[axis] = range(
1037        static_cast<ptrdiff_t>(boundaries[i]),
1038        static_cast<ptrdiff_t>(boundaries[i + 1]),
1039        std::nullopt);
1040      result.push_back(tensor[slices]);
1041    }
1042    return result;
1043  }

1045  template <typename T>
1046  std::vector<Tensor<T>> split(const Tensor<T>& tensor,
1047    size_t n_sections, size_t axis)
1048  {
1049    auto ndim = tensor.shape().size();
1050    if (axis >= ndim)
1051      throw std::invalid_argument("split: axis " + std::to_string(axis) +
1052        " out of range for " + std::to_string(ndim) + "-D tensor");
1053
1054    if (n_sections == 0)
1055      throw std::invalid_argument("split: number of sections must be > 0");
1056
1057    auto axis_size = tensor.shape()[axis];
1058    if (axis_size % n_sections != 0)
1059      throw std::invalid_argument("split: tensor of size " +
1060        std::to_string(axis_size) + " along axis " + std::to_string(axis) +
1061        " cannot be evenly split into " + std::to_string(n_sections) + " sections");
1062
1063    auto section_size = axis_size / n_sections;
1064    std::vector<size_t> split_points;
1065    split_points.reserve(n_sections - 1);
1066    for (size_t i = 1; i < n_sections; ++i)
1067      split_points.push_back(i * section_size);
1068
1069    return split(tensor, split_points, axis);
1070  }

1228  Tensor<T> Tensor<T>::reshape(const std::vector<ptrdiff_t>& new_shape) const
1229  {
1230    auto total = get_total_size();

1232    // Resolve -1 dimension
1233    std::vector<size_t> resolved(new_shape.size());

1235    size_t known_product = 1;
1236    for (size_t i = 0; i < new_shape.size(); ++i)
1237    {
1238      if (new_shape[i] == -1)
1239      {
1240        if (infer_idx >= 0)
1241          throw std::invalid_argument("Only one dimension can be inferred (-1)");

1243      }
1244      else if (new_shape[i] < 0)

1247      }

1258        throw std::invalid_argument("Cannot infer dimension: total size " +
1259          std::to_string(total) + " is not divisible by " + std::to_string(known_product));
1260      resolved[infer_idx] = total / known_product;
1261    }
1262
1263    // Validate total size
1264    size_t new_total = 1;
1265    for (auto d : resolved) new_total *= d;
1266    if (new_total != total)
1267      throw std::invalid_argument("Cannot reshape tensor of size " +
1268        std::to_string(total) + " into shape " + shape_to_string(resolved));
1269
1270    // Use contiguous source (copy if needed)
1271    Tensor<T> src = m_isContiguous ? *this : copy();
1272
1273    Tensor<T> ret;
1274    ret.m_data = src.m_data;

1287    }
