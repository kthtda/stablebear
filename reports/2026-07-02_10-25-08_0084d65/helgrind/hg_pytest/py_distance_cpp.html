153    template <typename TOperation>
154    static py::tuple cdist_impl(TensorT X, TensorT Y, TOperation op, [[maybe_unused]] CudaCdistFactory cudaFactory)
155    {
156      std::vector<size_t> outShape;
157      for (auto d : X.shape()) outShape.push_back(d);
158      for (auto d : Y.shape()) outShape.push_back(d);
159
160      auto outTensor = sb::Tensor<Tv>(outShape, Tv(0));
161
162      auto xTotal = X.size();
163      auto yTotal = Y.size();
164
165      if (xTotal == 0 || yTotal == 0)
166      {
167        std::unique_ptr<sb::StoppableTask<void>> empty_task = sb_py::execute_empty_task();
168        return py::make_tuple(std::move(empty_task), outTensor);
169      }
170
171#ifdef BUILD_WITH_CUDA
172      if (cudaFactory && !sb::settings().forceCpu
173          && xTotal * yTotal >= sb::settings().cudaThreshold * sb::settings().cudaThreshold)
174      {
175        if (sb::settings().deviceVerbose)
176        {
177          std::cout << "Cross-distance computation on CUDA device(s)" << std::endl;
178        }
179
180        auto rowPcfs = collect_all_pcfs(X);
181        auto colPcfs = collect_all_pcfs(Y);
182        auto task = cudaFactory(outTensor, rowPcfs, colPcfs, Tv(0), std::numeric_limits<Tv>::max());
183        task->start_async(sb::default_executor());
184        return py::make_tuple(std::move(task), outTensor);
185      }
186#endif
187
188      std::unique_ptr<sb::StoppableTask<void>> task =
189          sb_py::execute_stoppable_task<CdistTask<TOperation>>(X, Y, outTensor, op);
190      return py::make_tuple(std::move(task), outTensor);
191    }
