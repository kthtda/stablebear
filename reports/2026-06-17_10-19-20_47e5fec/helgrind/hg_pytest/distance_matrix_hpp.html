37  /// Lower-triangular compressed distance matrix (zero diagonal, nonnegative entries).
38  ///
39  /// Stores n*(n-1)/2 elements for an n×n symmetric matrix with
40  /// implicit zeros on the diagonal.
41  /// For i != j, element (i, j) maps to storage index
42  /// max(i,j)*(max(i,j)-1)/2 + min(i,j).
43  template <ArithmeticType T>
44  class DistanceMatrix
45  {
46  public:
47    using value_type = T;
48
49    class EntryProxy
50    {
51    public:
52      explicit EntryProxy(T* ptr) : m_ptr(ptr) { }
53
54      operator T() const
55      {
56        if (!m_ptr)
57          return T{};
58        return *m_ptr;
59      }
60
61      EntryProxy& operator=(const T& value)
62      {
63        if (value < T{})
64          throw std::invalid_argument("Distance matrix entries must be nonnegative");
65        if (!m_ptr)
66        {
67          if (value != T{})
68            throw std::invalid_argument("Diagonal entries of a distance matrix must be zero");
69          return *this;
70        }
71        *m_ptr = value;
72        return *this;
73      }
74
75    private:
76      T* m_ptr;
77    };
78
79    explicit DistanceMatrix(size_t n, const T& init = {})
80      : m_data(std::make_shared<T[]>(storage_size(n)))
81      , m_size(n)
82    {
83      if (init < T{})
84        throw std::invalid_argument("Distance matrix entries must be nonnegative");
85      std::fill(m_data.get(), m_data.get() + storage_size(n), init);
86    }

288      _M_invoke(const _Any_data& __functor, _ArgTypes&&... __args)
289      {
290	return std::__invoke_r<_Res>(*_Base::_M_get_pointer(__functor),
291				     std::forward<_ArgTypes>(__args)...);
292      }

90    [[nodiscard]] size_t size() const { return m_size; }
91    [[nodiscard]] size_t storage_count() const { return storage_size(m_size); }
92
93    [[nodiscard]] EntryProxy operator()(size_t i, size_t j)
94    {
95      bounds_check(i, j);
96      if (i == j)
97        return EntryProxy(nullptr);
98      return EntryProxy(&m_data[compressed_index(i, j)]);
99    }
100
101    [[nodiscard]] T operator()(size_t i, size_t j) const
102    {
103      bounds_check(i, j);
104      if (i == j)
105        return T{};
106      return m_data[compressed_index(i, j)];
107    }
108
109    [[nodiscard]] bool operator==(const DistanceMatrix& rhs) const
110    {
111      if (m_size != rhs.m_size)
112        return false;
113      return std::equal(m_data.get(), m_data.get() + storage_size(m_size), rhs.m_data.get());
114    }
115
116    [[nodiscard]] bool operator!=(const DistanceMatrix& rhs) const
117    {
118      if (m_size != rhs.m_size)
119        return true;
120      return std::mismatch(m_data.get(), m_data.get() + storage_size(m_size), rhs.m_data.get()).first
121        != m_data.get() + storage_size(m_size);
122    }
123
124    [[nodiscard]] const T* data() const { return m_data.get(); }
125
126    [[nodiscard]] static size_t storage_size(size_t n)
127    {
128      return n * (n - 1) / 2;
129    }

131  private:
132    [[nodiscard]] T* mutable_data() { return m_data.get(); }

135    friend MatT io::detail::read_compressed_matrix(std::istream&);

137    void bounds_check(size_t i, size_t j) const
138    {
139      if (i >= m_size || j >= m_size)
140      {
141        std::ostringstream oss;
142        oss << "DistanceMatrix index (" << i << ", " << j << ") out of range for matrix of size " << m_size;
143        throw std::out_of_range(oss.str());
144      }
145    }
146
147    [[nodiscard]] static size_t compressed_index(size_t i, size_t j)
148    {
149      auto row = std::max(i, j);
150      auto col = std::min(i, j);
151      return row * (row - 1) / 2 + col;
152    }

155    size_t m_size;
