352  Tensor<T> Tensor<T>::broadcast_to(const std::vector<size_t>& target_shape) const
353  {
354    size_t ndim = target_shape.size();
355    size_t src_ndim = m_shape.size();
356
357    if (ndim < src_ndim)
358    {
359      throw std::invalid_argument("Cannot broadcast shape " +
360          shape_to_string(m_shape) + " to " + shape_to_string(target_shape) +
361          ": target has fewer dimensions");
362    }
363
364    Tensor ret;
365    ret.m_data = m_data;
366    ret.m_offset = m_offset;
367    ret.m_shape = target_shape;
368    ret.m_strides.resize(ndim);
369    ret.m_isContiguous = false;
370
371    size_t prepended = ndim - src_ndim;
372    for (size_t i = 0; i < ndim; ++i)
373    {
374      if (i < prepended)
375      {
376        ret.m_strides[i] = 0;
377      }
378      else
379      {
380        size_t si = i - prepended;
381        if (m_shape[si] == target_shape[i])
382        {
383          ret.m_strides[i] = m_strides[si];
384        }
385        else if (m_shape[si] == 1)
386        {
387          ret.m_strides[i] = 0;
388        }
389        else
390        {
391          throw std::invalid_argument("Cannot broadcast shape " +
392              shape_to_string(m_shape) + " to " + shape_to_string(target_shape));
393        }
394      }
395    }
396
397    return ret;
398  }

406    template <typename T, typename BinaryOp,

413      Tensor<R> result(out_shape);

415        result(idx) = op(lhs_view(idx), rhs_view(idx));

508  template <typename T>
509  Tensor<T> masked_select(const Tensor<T>& src, const Tensor<bool>& mask)
510  {
511    if (mask.shape() != src.shape())
512    {
513      throw std::invalid_argument(
514        "masked_select: mask shape does not match tensor shape");
515    }
516
517    // Single pass: collect matching elements into a buffer
518    std::vector<T> buf;
519    walk(src, [&](const std::vector<size_t>& idx) {
520      if (mask(idx))
521        buf.push_back(src(idx));
522    });
523
524    if (buf.empty())
525    {
526      return Tensor<T>({0});
527    }
528
529    Tensor<T> result({buf.size()});
530    std::move(buf.begin(), buf.end(), result.data());
531
532    return result;
533  }

763    struct ResolvedSelector
764    {
765      size_t axis;
766      std::vector<size_t> indices;  // selected positions along this axis
767    };
768
769    inline std::vector<ResolvedSelector> resolve_selectors(
770      const std::vector<std::pair<size_t, AxisSelector>>& selectors,
771      const std::vector<size_t>& shape)
772    {
773      std::vector<ResolvedSelector> result;
774      result.reserve(selectors.size());
775      for (const auto& [axis, sel] : selectors)
776      {
777        if (axis >= shape.size())
778          throw std::invalid_argument("outer indexing: axis " + std::to_string(axis) +
779            " out of range for tensor with " + std::to_string(shape.size()) + " dimensions");
780
781        ResolvedSelector rs;
782        rs.axis = axis;
783        std::visit([&](const auto& idx_tensor) {
784          using IdxT = std::decay_t<decltype(idx_tensor)>;
785          if (idx_tensor.shape().size() != 1)
786            throw std::invalid_argument("outer indexing: selector must be 1D");
787
788          if constexpr (std::is_same_v<IdxT, Tensor<bool>>)
789          {
790            if (idx_tensor.shape()[0] != shape[axis])
791              throw std::invalid_argument("outer indexing: bool mask length must match axis size");
792            rs.indices = collect_true_indices(idx_tensor);
793          }
794          else
795          {
796            const auto axis_size = static_cast<ptrdiff_t>(shape[axis]);
797            rs.indices.reserve(idx_tensor.shape()[0]);
798            for (size_t i = 0; i < idx_tensor.shape()[0]; ++i)
799            {
800              // Resolve negatives against the axis size and bounds-check so
801              // the C++ entry point is memory-safe regardless of caller.
802              auto v = static_cast<ptrdiff_t>(idx_tensor({i}));
803              if (v < 0) v += axis_size;
804              if (v < 0 || v >= axis_size)
805                throw std::out_of_range(
806                  "index " + std::to_string(static_cast<ptrdiff_t>(idx_tensor({i}))) +
807                  " is out of bounds for axis with size " + std::to_string(axis_size));
808              rs.indices.push_back(static_cast<size_t>(v));
809            }
810          }
811        }, sel);
812        result.push_back(std::move(rs));
813      }
814      return result;
815    }
816  }

922  template <typename T>
923  Tensor<T> concatenate(const std::vector<Tensor<T>>& tensors, size_t axis)
924  {
925    if (tensors.empty())
926      throw std::invalid_argument("concatenate: need at least one tensor");
927
928    auto ndim = tensors[0].shape().size();
929    if (axis >= ndim)
930      throw std::invalid_argument("concatenate: axis " + std::to_string(axis) +
931        " out of range for " + std::to_string(ndim) + "-D tensor");
932
933    // Validate shapes match on all axes except the join axis
934    for (size_t t = 1; t < tensors.size(); ++t)
935    {
936      if (tensors[t].shape().size() != ndim)
937        throw std::invalid_argument("concatenate: all tensors must have the same number of dimensions");
938      for (size_t d = 0; d < ndim; ++d)
939      {
940        if (d != axis && tensors[t].shape()[d] != tensors[0].shape()[d])
941          throw std::invalid_argument("concatenate: shape mismatch on dimension " + std::to_string(d));
942      }
943    }
944
945    // Compute output shape
946    auto out_shape = tensors[0].shape();
947    out_shape[axis] = 0;
948    for (const auto& t : tensors)
949      out_shape[axis] += t.shape()[axis];
950
951    Tensor<T> result(out_shape);
952
953    // Copy each tensor's data into the result
954    size_t offset = 0;
955    for (const auto& src : tensors)
956    {
957      auto src_axis_size = src.shape()[axis];
958      walk(src, [&](const std::vector<size_t>& src_idx) {
959        auto dst_idx = src_idx;
960        dst_idx[axis] += offset;
961        result(dst_idx) = src(src_idx);
962      });
963      offset += src_axis_size;
964    }
965
966    return result;
967  }

969  template <typename T>
970  Tensor<T> stack(const std::vector<Tensor<T>>& tensors, ptrdiff_t axis)
971  {
972    if (tensors.empty())
973      throw std::invalid_argument("stack: need at least one tensor");
974
975    auto ndim = static_cast<ptrdiff_t>(tensors[0].shape().size());
976
977    // Resolve negative axis; valid range for new axis is [-(ndim+1), ndim]
978    if (axis < 0)
979      axis += ndim + 1;
980    if (axis < 0 || axis > ndim)
981      throw std::invalid_argument("stack: axis out of range");
982
983    // Validate all shapes match
984    for (size_t t = 1; t < tensors.size(); ++t)
985    {
986      if (tensors[t].shape() != tensors[0].shape())
987        throw std::invalid_argument("stack: all tensors must have the same shape");
988    }
989
990    // expand_dims each tensor, then concatenate along the new axis
991    std::vector<Tensor<T>> expanded;
992    expanded.reserve(tensors.size());
993    for (const auto& t : tensors)
994      expanded.push_back(t.expand_dims(axis));
995
996    return concatenate(expanded, static_cast<size_t>(axis));
997  }

999  template <typename T>
1000  std::vector<Tensor<T>> split(const Tensor<T>& tensor,
1001    const std::vector<size_t>& split_points, size_t axis)
1002  {
1003    auto ndim = tensor.shape().size();
1004    if (axis >= ndim)
1005      throw std::invalid_argument("split: axis " + std::to_string(axis) +
1006        " out of range for " + std::to_string(ndim) + "-D tensor");
1007
1008    auto axis_size = tensor.shape()[axis];
1009
1010    // Build slice boundaries: [0, sp0, sp1, ..., axis_size]
1011    std::vector<size_t> boundaries;
1012    boundaries.reserve(split_points.size() + 2);
1013    boundaries.push_back(0);
1014    for (auto sp : split_points)
1015    {
1016      if (sp > axis_size)
1017        sp = axis_size;
1018      boundaries.push_back(sp);
1019    }
1020    boundaries.push_back(axis_size);
1021
1022    std::vector<Tensor<T>> result;
1023    result.reserve(boundaries.size() - 1);
1024    for (size_t i = 0; i + 1 < boundaries.size(); ++i)
1025    {
1026      std::vector<Slice> slices(ndim, all());
1027      slices[axis] = range(
1028        static_cast<ptrdiff_t>(boundaries[i]),
1029        static_cast<ptrdiff_t>(boundaries[i + 1]),
1030        std::nullopt);
1031      result.push_back(tensor[slices]);
1032    }
1033    return result;
1034  }

1038    size_t n_sections, size_t axis)
1039  {

1043        " out of range for " + std::to_string(ndim) + "-D tensor");

1046    if (axis_size % n_sections != 0)
1047      throw std::invalid_argument("split: tensor of size " +

1049        " cannot be evenly split into " + std::to_string(n_sections) + " sections");
1050

1052    std::vector<size_t> split_points;
1053    split_points.reserve(n_sections - 1);
1054    for (size_t i = 1; i < n_sections; ++i)
1055      split_points.push_back(i * section_size);

1057    return split(tensor, split_points, axis);

1060  template <typename T>
1061  std::vector<Tensor<T>> array_split(const Tensor<T>& tensor,
1062    size_t n_sections, size_t axis)
1063  {
1064    auto ndim = tensor.shape().size();
1065    if (axis >= ndim)
1066      throw std::invalid_argument("array_split: axis " + std::to_string(axis) +
1067        " out of range for " + std::to_string(ndim) + "-D tensor");
1068    if (n_sections == 0)
1069      throw std::invalid_argument("array_split: number of sections must be > 0");
1070
1071    auto axis_size = tensor.shape()[axis];
1072    auto base_size = axis_size / n_sections;
1073    auto remainder = axis_size % n_sections;
1074
1075    std::vector<size_t> split_points;
1076    split_points.reserve(n_sections - 1);
1077    size_t pos = 0;

1215  template <typename T>
1216  Tensor<T> Tensor<T>::reshape(const std::vector<ptrdiff_t>& new_shape) const
1217  {
1218    auto total = get_total_size();
1219
1220    // Resolve -1 dimension
1221    std::vector<size_t> resolved(new_shape.size());
1222    ptrdiff_t infer_idx = -1;
1223    size_t known_product = 1;
1224    for (size_t i = 0; i < new_shape.size(); ++i)
1225    {
1226      if (new_shape[i] == -1)
1227      {
1228        if (infer_idx >= 0)
1229          throw std::invalid_argument("Only one dimension can be inferred (-1)");
1230        infer_idx = static_cast<ptrdiff_t>(i);
1231      }
1232      else if (new_shape[i] < 0)
1233      {
1234        throw std::invalid_argument("Invalid dimension size: " + std::to_string(new_shape[i]));
1235      }
1236      else
1237      {
1238        resolved[i] = static_cast<size_t>(new_shape[i]);
1239        known_product *= resolved[i];
1240      }
1241    }
1242
1243    if (infer_idx >= 0)
1244    {
1245      if (known_product == 0 || total % known_product != 0)
1246        throw std::invalid_argument("Cannot infer dimension: total size " +
1247          std::to_string(total) + " is not divisible by " + std::to_string(known_product));
1248      resolved[infer_idx] = total / known_product;
1249    }
1250
1251    // Validate total size
1252    size_t new_total = 1;
1253    for (auto d : resolved) new_total *= d;
1254    if (new_total != total)
1255      throw std::invalid_argument("Cannot reshape tensor of size " +
1256        std::to_string(total) + " into shape " + shape_to_string(resolved));
1257
1258    // Use contiguous source (copy if needed)
1259    Tensor<T> src = m_isContiguous ? *this : copy();
1260
1261    Tensor<T> ret;
1262    ret.m_data = src.m_data;
1263    ret.m_offset = src.m_offset;
1264    ret.m_shape = resolved;
1265    ret.m_isContiguous = true;
1266    ret.m_viewType = ViewType::Base;
1267
1268    // Compute row-major strides
1269    ret.m_strides.resize(resolved.size());
1270    if (!resolved.empty())
1271    {
1272      ret.m_strides.back() = 1;
1273      for (auto i = static_cast<ptrdiff_t>(resolved.size()) - 2; i >= 0; --i)
1274        ret.m_strides[i] = ret.m_strides[i + 1] * static_cast<ptrdiff_t>(resolved[i + 1]);
1275    }
1276
1277    return ret;
1278  }

1461  template <typename T>
1462  template <typename SliceVector>
1463  Tensor<T> Tensor<T>::extract(SliceVector sliceVector) const
1464  {
1465    Tensor ret;
1466
1467    ret.m_data = m_data; // view
1468
1469    // temporary just to get sizes
1470    ret.m_shape = m_shape;
1471    ret.m_strides = m_strides;
1472
1473    ret.m_offset = m_offset;
1474
1475    std::vector<size_t> dimsToDrop;
1476
1477    // For now, we'll drop the assumption that the tensor is contiguous in memory
1478    // as soon as we extract a subtensor. There are, however, some cases where the resulting subtensor would be
1479    // contiguous that we can try to optimize for in the future (e.g., extracting the top n rows of a matrix).
1480    // Things will work fine with this assumption dropped but certain operations could be a little slower (probably
1481    // unlikely to matter for the type of things we're targeting).
1482    ret.m_isContiguous = false;
1483
1484    size_t i = 0;
1485    for (auto & slice : sliceVector)
1486    {
1487      std::visit([i, &ret, &dimsToDrop](auto&& arg) {
1488        using argT = std::decay_t<decltype(arg)>;
1489        if constexpr (std::is_same_v<argT, SliceIndex>)
1490        {
1491          // Resolve a (possibly negative) single integer index against the
1492          // dimension size and bounds-check it, mirroring NumPy semantics.
1493          // Throwing here (rather than computing an out-of-buffer offset)
1494          // turns silent OOB reads / SIGSEGV-on-write into a clean IndexError.
1495          auto dim_size = static_cast<ptrdiff_t>(ret.m_shape[i]);
1496          auto ix = arg.index;
1497          if (ix < 0) ix += dim_size;
1498          if (ix < 0 || ix >= dim_size)
1499          {

1501              "index " + std::to_string(arg.index) +
1502              " is out of bounds for axis " + std::to_string(i) +
1503              " with size " + std::to_string(dim_size));
1504          }
1505          ret.m_shape[i] = 1;
1506          ret.m_offset += ix * ret.m_strides[i];

1508        }

1510        {
1511          if (!arg.step)
1512          {
1513            arg.step = 1_z;
1514          }

1524          // Mirror Python's slice.indices(n): resolve negative bounds by
1525          // adding dim_size, then clamp into the valid range. For a positive
1526          // step the range is [0, n]; for a negative step it is [-1, n-1].
1527          const ptrdiff_t lower = (step < 0) ? -1_z : 0_z;
1528          const ptrdiff_t upper = (step < 0) ? dim_size - 1_z : dim_size;
1529
1530          ptrdiff_t start;
1531          if (!arg.start)

1542            }
1543            else if (start > upper)

1547          }

1554          else
1555          {
1556            stop = *arg.stop;
1557            if (stop < 0)
1558            {
1559              stop += dim_size;
1560              if (stop < lower) stop = lower;
1561            }
1562            else if (stop > upper)
1563            {
1564              stop = upper;
1565            }
1566          }
1567
1568          ptrdiff_t len;
1569          if (step > 0)
1570          {
1571            len = (start < stop) ? (stop - start - 1_z) / step + 1_z : 0_z;
1572          }
1573          else
1574          {
1575            len = (start > stop) ? (start - stop - 1_z) / (-step) + 1_z : 0_z;
1576          }
1577
1578          ret.m_shape[i] = static_cast<size_t>(len);
1579          ret.m_offset += start * ret.m_strides[i];
1580          ret.m_strides[i] *= step;
1581        }
1582        // For SliceAll, don't modify shape
1583      }, slice);
1584      ++i;
1585    }
1586
1587    size_t nDroppedDims = 0_uz;
1588    for (auto dim : dimsToDrop)
1589    {

1591      ret.m_strides.erase(ret.m_strides.begin() + dim - nDroppedDims);
1592      ++nDroppedDims;
1593    }
1594
1595    return ret;
1596  }
