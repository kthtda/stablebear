97  /// CPU cross-distance integration task (cdist).
98  /// Computes all (i,j) pairs between two separate function sets.
99  template <typename TOperation, typename PcfFwdIt>
100  class CpuCrossIntegrationTask : public sb::StoppableTask<void>
101  {
102  public:
103    using pcf_type = typename PcfFwdIt::value_type;
104    using value_type = typename pcf_type::value_type;
105    using time_type = typename pcf_type::time_type;
106
107    CpuCrossIntegrationTask(Tensor<value_type> out,
108                 PcfFwdIt beginRows, PcfFwdIt endRows,
109                 PcfFwdIt beginCols, PcfFwdIt endCols,
110                 TOperation op)
111      : m_rowFs(beginRows, endRows)
112      , m_colFs(beginCols, endCols)
113      , m_out(std::move(out))
114      , m_op(op)
115    { }
116
117  private:
118    tf::Future<void> run_async(Executor& exec) override
119    {
120      auto nRows = m_rowFs.size();
121      auto nCols = m_colFs.size();
122
123      next_step(nRows * nCols, "Computing cross-distances.", "integral");
124
125      tf::Taskflow flow;
126      if (nRows == 0 || nCols == 0)
127      {
128        return exec.cpu()->run(std::move(flow));
129      }
130
131      value_type* data = m_out.data();
132      flow.for_each_index<size_t, size_t, size_t>(0ul, nRows, 1ul, [this, data, nCols](size_t i) {
133        if (stop_requested())
134        {
135          return;
136        }
137        for (size_t j = 0; j < nCols; ++j)
138        {
139          data[i * nCols + j] = m_op(integrate<time_type, value_type>(
140              m_rowFs[i], m_colFs[j], m_op, 0, std::numeric_limits<time_type>::max()));
141        }
142        add_progress(nCols);
143      });
144
145      return exec.cpu()->run(std::move(flow));
146    }
147
148    std::vector<pcf_type> m_rowFs;
149    std::vector<pcf_type> m_colFs;
150    Tensor<value_type> m_out;
151    TOperation m_op;
152  };
